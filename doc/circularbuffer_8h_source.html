57 virtual uint16_t push(const T* data, uint16_t size)
58 {
59 if (size > availableSpace())
60 size = availableSpace();
61
62 uint16_t writeCount = 0;
63
64 while (writeCount < size) {
65 _buffer[_writeHead] = data[writeCount++];
66 incrementOrResetHead(_writeHead);

94 virtual uint16_t pull(T* data, uint16_t size)
95 {
96 if (size > availableData())
97 size = availableData();
98
99 uint16_t readCount = 0;
100
101 while (readCount < size) {
102 data[readCount++] = _buffer[_readHead];
103 incrementOrResetHead(_readHead);
104 }
105 _availableData -= readCount;
106
107 return readCount;
108 }

116 virtual T pull()
117 {
118 T data = _buffer[_readHead];
119 incrementOrResetHead(_readHead);
120 if (_availableData > 0)
121 _availableData--;
122
123 return data;
124 }
