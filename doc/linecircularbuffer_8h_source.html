38template <uint16_t BUFFER_SIZE>
39class LineCircularBuffer : public CircularBuffer<char, BUFFER_SIZE>
40{

42 LineCircularBuffer() :

46 uint16_t push(const char* data, uint16_t size) override
47 {
48 if (size > CircularBuffer<char, BUFFER_SIZE>::availableSpace())
49 size = CircularBuffer<char, BUFFER_SIZE>::availableSpace();
50
51 uint16_t writeCount = 0;
52
53 while (writeCount < size) {
54 push(data[writeCount++]);
55 }
56
57 return writeCount;
58 }

60 void push(char data) override
61 {

63
64 if (data == '\n') {
65 _bufferedLines++;
66 }
67 }

69 virtual uint16_t pull(char* data, uint16_t size) override
70 {
71 if (size > CircularBuffer<char, BUFFER_SIZE>::availableData())
72 size = CircularBuffer<char, BUFFER_SIZE>::availableData();
73
74 uint16_t readCount = 0;
75
76 while (readCount < size) {
77 data[readCount++] = pull();
78 }
79
80 return readCount;
81 }
82
83 char pull() override
84 {
85 char data = CircularBuffer<char, BUFFER_SIZE>::pull();
86
87 if (data == '\n') {
88 _bufferedLines--;
89 }
90
91 return data;
92 }
93
97 inline uint16_t numBufferedLines() const
98 {
99 return _bufferedLines;
100 }
101
108 uint16_t readLine(char* data, uint16_t size)
109 {
110 uint16_t readCount = 0;
111 char c = '\0';
112
113 while (!CircularBuffer<char, BUFFER_SIZE>::isEmpty() && c != '\n') {
114 c = pull();
115 if (readCount < size) {
116 data[readCount++] = c;
117 }
118 }
119
120 return readCount;
121 }

123 private:
